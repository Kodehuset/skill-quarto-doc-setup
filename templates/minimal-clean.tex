\newcommand{\makeinlinetitle}{

  $endif$

  $if(author)$
  {\normalsize $author$}%
  $endif$
  $if(organization)$
  $if(author)$ — $endif${\normalsize $organization$}%
  $endif$
  $if(author)$\par$else$$if(organization)$\par$endif$$endif$
  \vspace{0.5cm}
  \rule{\textwidth}{0.4pt}
  \vspace{0.5cm}
}
