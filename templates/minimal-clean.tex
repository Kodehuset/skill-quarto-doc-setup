% ============================================================================
% INLINE TITLE (no separate title page)
% ============================================================================

\newcommand{\makeinlinetitle}{
  \vspace*{1cm}
  {\Huge\bfseries $title$\par}
  $if(subtitle)$
  \vspace{0.3cm}
  {\Large\color{textgray} $subtitle$\par}
  $endif$
  \vspace{0.3cm}
  $if(author)$
  {\normalsize $author$}%
  $endif$
  $if(organization)$
  $if(author)$ — $endif${\normalsize $organization$}%
  $endif$
  $if(author)$\par$else$$if(organization)$\par$endif$$endif$
  \vspace{0.5cm}
  \rule{\textwidth}{0.4pt}
  \vspace{0.5cm}
}
