  $endif$

  $if(author)$
  {\normalsize $author$}%
  $endif$
  $if(organization)$
  $if(author)$ — $endif${\normalsize $organization$}%
  $endif$
  $if(author)$\par$else$$if(organization)$\par$endif$$endif$
  $if(date)$
  {\small\textcolor{textgray}{$date$}\par}
  $endif$
  \vspace{0.5cm}
  \rule{\textwidth}{0.4pt}
  \vspace{0.5cm}
}
