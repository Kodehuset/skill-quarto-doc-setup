\newcommand{\maketitlepage}{
  \begin{titlepage}
    \centering
    \vspace*{2cm}

    $if(logo)$
    \includegraphics[width=0.3\textwidth]{$logo$}\\[1.5cm]
    $else$
    \vspace{1.5cm}
    $endif$

    {\Huge\bfseries\color{accent} $title$\par}
    \vspace{0.5cm}
    $if(subtitle)$
    {\Large\color{textgray} $subtitle$\par}
    \vspace{1cm}
    $endif$

    \rule{0.6\textwidth}{0.4pt}\\[1.5cm]

    $if(version)$
    {\large Version: \textbf{$version$}\par}
    \vspace{0.3cm}
    $endif$

    \vfill

    $if(organization)$
    {\large\textbf{$organization$}\par}
    \vspace{0.3cm}
    $endif$

    $if(author)$
    {\normalsize Prepared by: $author$\par}
    $endif$

    $if(document_status)$
    \vspace{1cm}
    {\small\textcolor{textgray}{Document Status: $document_status$}\par}
    $endif$

  \end{titlepage}
}
